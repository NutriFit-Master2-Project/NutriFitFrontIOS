NavigationStack {
    List {
        NavigationLink("Voir les détails de l'article", value: "ArticleDetails")
        NavigationLink("Voir le profil", value: "UserProfile")
    }
    .navigationTitle("Menu")
    .navigationDestination(for: String.self) { valeur in
        switch valeur {
        case "ArticleDetails":
            ArticleDetailsView()
        case "UserProfile":
            UserProfileView()
        default:
            Text("Vue non trouvée")
        }
    }
}
